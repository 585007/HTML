function simple_gui2
n = 0;
f = figure('Position',[300,300,900,600]);
h = [];
h(1) = uicontrol('Style','popupmenu','String',{'Peaks','Membrane','Sinc','Dalen','Spesiell'},'Position',[690,20,200,50],'Callback',@popup_menu_Callback);
h(2) = axes('Units','pixels','Position',[50,40,510,510]);
h(3) = uicontrol('Style','pushbutton','String','Rotate','Position',[690,80,200,50],'Callback',@rot_Callback);
h(4) = uicontrol('Style','pushbutton','String','Top view','Position',[690,140,200,50],'Callback',@top_Callback);
h(5) = uicontrol('Style','pushbutton','String','Left','Position',[690,200,100,50],'Callback',@left_Callback);
h(6) = uicontrol('Style','pushbutton','String','Right','Position',[790,200,100,50],'Callback',@right_Callback);
h(7) = uicontrol('Style','pushbutton','String','Start','Position',[690,260,100,50],'Callback',@start_Callback);
h(8) = uicontrol('Style','pushbutton','String','Stopp','Position',[790,260,100,50],'Callback',@stopp_Callback);

tmr = timer ('ExecutionMode','FixedRate','Period', 1.01, 'TimerFcn', {@rot_Callback});

f.Units = 'normalized';
set(h, 'Units', 'normalized');
set(h, 'FontSize', 12);

peaks_data = peaks(35);
membrane_data = membrane;
[x,y] = meshgrid(-15:.5:15);
z = sqrt(x.^2+y.^2) + eps;
sinc_data = sin(z)./z;
rarFig_data = sin(z)./z + tan(z);
t = 0:pi/10:2*pi;
sylinder_data = cylinder(2+cos(t));

[y, Fs] = audioread('engine.wav');

player = audioplayer(y, Fs);

current_data = peaks_data;
s = surf(current_data);

function popup_menu_Callback(source,eventdata)
str = get(source, 'String');
val = get(source,'Value');

switch str{val};
case 'Peaks' % User selects Peaks.
current_data = peaks_data;
x = [-2 42];
y = [-2 42];
z = [-10 20];
case 'Membrane' current_data = membrane_data;
x = [-4 36];
y = [-4 36];
z = [-2 4];
case 'Sinc'
current_data = sinc_data;
x = [-10 70];
y = [-10 70];
z = [-10 20];
case 'Spesiell'
current_data = rarFig_data;
x = [-10 70];
y = [-10 70];
z = [-230 50];
case 'Dalen'
current_data = sylinder_data;
x = [-1 25];
y = [-1 25];
z = [-10 20];
end
s = surf(current_data);
xlim(x);
ylim(y);
zlim(z);

end
function rot_Callback(obj, ev)
rotate(s, [0 0 1], 5);
direction = [0 0 1];
rotate(s, direction, n);
n = n + 1;
end
function top_Callback(obj, ev)
view(2);
end
function right_Callback(obj, ev)
view(0,0);
view(90,0);
end
function left_Callback(obj, ev)
view(0,0);
view(-90,0);
end
function start_Callback(obj, ev)
play(player);

start(tmr);
end
function stopp_Callback(obj, ev)
pause(player);

stop(tmr);
end
xlim([-2 42]);
ylim([-2 42]); zlim([-10 20]);
end
